# Stack VM + LLM Codegen Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace hardcoded float-array genomes with evolved opcode-list programs executed by a stack VM, and add an LLM codegen loop that generates new primitives from telemetry every 10 generations.

**Architecture:** A `StackVM` in `shared/vm/` executes JSON opcode lists produced by the GA. `instruction_set.py` owns all SENSE/DO handler dispatch and auto-loads LLM-generated plugins from `shared/vm/generated/`. Red and Blue agents become thin shells: load program from `round_config.json`, populate an `AgentContext`, call `vm.run()` each tick. The GA engine, coevolution engine, and run_simulation are updated to produce and evolve opcode lists instead of float arrays.

**Tech Stack:** Python 3.10, psutil, numpy, ctypes, winreg, anthropic SDK (codegen only)

```python
MAX_PROGRAM_LEN = 64

_OPERAND_OPCODES = {"PUSH", "LOOP"}


def compute_operand_mask(program: list) -> list:
    mask = [False] * len(program)
    for i, op in enumerate(program):
        if op in _OPERAND_OPCODES and i + 1 < len(program):
            mask[i + 1] = True
    return mask
```

```python
def repair_control_flow(program: list) -> list:
    mask = compute_operand_mask(program)

    # Pass 1: strip orphaned ELSE/ENDIF
    result = []
    if_depth = 0
    for i, op in enumerate(program):
        if mask[i]:
            result.append(op)
            continue
        if op == "IF_TRUE":
            if_depth += 1
            result.append(op)
        elif op == "ELSE":
            if if_depth > 0:
                result.append(op)
        elif op == "ENDIF":
            if if_depth > 0:
                if_depth -= 1
                result.append(op)
        else:
            result.append(op)

    # Pass 2: close unclosed IF_TRUE blocks
    for _ in range(if_depth):
        if len(result) < MAX_PROGRAM_LEN:
            result.append("ENDIF")

    # Pass 3: strip orphaned ENDLOOP
    final = []
    loop_depth = 0
    i = 0
    while i < len(result):
        op = result[i]
        if op == "LOOP":
            loop_depth += 1
            final.append(op)
            i += 1
            if i < len(result):
                final.append(result[i])
        elif op == "ENDLOOP":
            if loop_depth > 0:
                loop_depth -= 1
                final.append(op)
        else:
            final.append(op)
        i += 1

    # Pass 4: close unclosed LOOP blocks
    for _ in range(loop_depth):
        if len(final) < MAX_PROGRAM_LEN:
            final.append("ENDLOOP")

    return final[:MAX_PROGRAM_LEN]
```

```python
def _find_else_or_endif(program, ip, mask):
    depth = 0
    i = ip + 1
    while i < len(program):
        if not mask[i]:
            if program[i] == "IF_TRUE":
                depth += 1
            elif program[i] == "ELSE" and depth == 0:
                return i
            elif program[i] == "ENDIF":
                if depth == 0:
                    return i
                depth -= 1
        i += 1
    return len(program)


def _find_endif(program, ip, mask):
    depth = 0
    i = ip + 1
    while i < len(program):
        if not mask[i]:
            if program[i] == "IF_TRUE":
                depth += 1
            elif program[i] == "ENDIF":
                if depth == 0:
                    return i
                depth -= 1
        i += 1
    return len(program)


_SENSE_MAP = {
    "SENSE_DELTA":           ("current_delta",        "both"),
    "SENSE_TIER":            ("current_tier",          "both"),
    "SENSE_PROC_COUNT":      ("snap_proc_count",       "both"),
    "SENSE_CPU_AVG":         ("snap_cpu_avg",          "both"),
    "SENSE_TICK":            ("current_tick",          "both"),
    "SENSE_SCHED_TASKS":     ("snap_sched_tasks",      "both"),
    "SENSE_OUTBOUND_CONNS":  ("snap_outbound_conns",   "both"),
    "SENSE_CHILD_DEPTH":     ("snap_child_depth",      "both"),
    "SENSE_PRIVESC_SIGNALS": ("snap_privesc_signals",  "both"),
    "SENSE_PORT_COUNT":      ("snap_port_count",       "both"),
    "SENSE_BLUE_RESPONSES":  ("snap_blue_responses",   "red"),
    "SENSE_NEW_PROCS":       ("snap_new_procs",        "blue"),
}


def _do_sense(op, agent):
    entry = _SENSE_MAP.get(op)
    if entry is None:
        return 0.0
    attr, role_gate = entry
    if role_gate != "both" and getattr(agent, "role", "red") != role_gate:
        return 0.0
    return float(getattr(agent, attr, 0.0))


class StackVM:
    def run(self, program: list, agent) -> None:
        from shared.vm import instruction_set
        mask = compute_operand_mask(program)
        stack: list = []
        loop_stack: list = []
        ip = 0
        steps = 0

        while ip < len(program) and steps < MAX_STEPS:
            if mask[ip]:
                ip += 1
                continue

            op = program[ip]
            steps += 1

            if op == "PUSH":
                ip += 1
                stack.append(float(program[ip]) if ip < len(program) else 0.0)
            elif op == "POP":
                if stack:
                    stack.pop()
            elif op == "DUP":
                stack.append(stack[-1] if stack else 0.0)
            elif op == "ADD":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(a + b)
            elif op == "SUB":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(a - b)
            elif op == "MUL":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(a * b)
            elif op == "GT":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(1.0 if a > b else 0.0)
            elif op == "LT":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(1.0 if a < b else 0.0)
            elif op == "GTE":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(1.0 if a >= b else 0.0)
            elif op == "LTE":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(1.0 if a <= b else 0.0)
            elif op == "EQ":
                b = stack.pop() if stack else 0.0
                a = stack.pop() if stack else 0.0
                stack.append(1.0 if a == b else 0.0)
            elif op == "IF_TRUE":
                cond = stack.pop() if stack else 0.0
                if cond <= 0.5:
                    ip = _find_else_or_endif(program, ip, mask)
            elif op == "ELSE":
                ip = _find_endif(program, ip, mask)
            elif op == "ENDIF":
                pass
            elif op == "LOOP":
                ip += 1
                raw = program[ip] if ip < len(program) else 1
                n = max(1, min(8, int(float(raw))))
                if len(loop_stack) < 4:
                    loop_stack.append((ip, n))
                # 5th nested loop → NOP (don't push)
            elif op == "ENDLOOP":
                if loop_stack:
                    ret_ip, remaining = loop_stack[-1]
                    if remaining > 1:
                        loop_stack[-1] = (ret_ip, remaining - 1)
                        ip = ret_ip
                    else:
                        loop_stack.pop()
            elif op == "HALT":
                break
            elif op == "NOP":
                pass
            elif op.startswith("SENSE_"):
                stack.append(_do_sense(op, agent))
            elif op.startswith("DO_"):
                instruction_set.dispatch(op, agent)
            # unknown → NOP

            ip += 1
```

```python
    if not _GENERATED_DIR.exists():
        return

        if py_file.name == "__init__.py":
            continue
        source = py_file.read_text()
        import re
        m = re.search(r"^# OPCODE: (DO_\w+|SENSE_\w+)", source, re.MULTILINE)
        fn_m = re.search(r"^def (do_\w+|sense_\w+)\(agent\)", source, re.MULTILINE)
        if not (m and fn_m):
            _LOG.error("Plugin %s missing header — skipped", py_file.name)
            continue
        opcode = m.group(1)
        if opcode in _BUILTINS:
            _LOG.error("Plugin %s collides with builtin %s — skipped", py_file.name, opcode)
            continue
        spec = importlib.util.spec_from_file_location(py_file.stem, py_file)
        mod = importlib.util.module_from_spec(spec)
        try:
            spec.loader.exec_module(mod)
        except Exception as e:
            _LOG.error("Plugin %s load error: %s — skipped", py_file.name, e)
            continue
        fn_name = fn_m.group(1)
        fn = getattr(mod, fn_name, None)
        if fn is None:
            _LOG.error("Plugin %s: function %s not found — skipped", py_file.name, fn_name)
            continue
        _registry[opcode] = fn
        _LOG.info("Loaded plugin op %s from %s", opcode, py_file.name)


_load_plugins()
```

```python
import collections
import numpy as np  # already imported at top

NOISE_SETTLE_TICKS = 10


class AdaptiveNoiseFloor:
    def __init__(self, window: int = 20):
        self._samples: collections.deque = collections.deque(maxlen=window)
        self._nominal_streak: int = 0
        self._ticks_since_recal: int = 0
        self.floor: float = 0.0

    def seed(self, samples: list) -> None:
        for s in samples:
            self._samples.append(s)
        self.floor = float(np.mean(self._samples)) if self._samples else 0.0

    def update(self, delta_raw: float, tier: str,
               settle_ticks: int = NOISE_SETTLE_TICKS) -> None:
        self._samples.append(delta_raw)
        self._ticks_since_recal += 1
        if tier == "NOMINAL":
            self._nominal_streak += 1
        else:
            self._nominal_streak = 0
        if self._nominal_streak >= settle_ticks:
            self._recalibrate()
        elif self._ticks_since_recal >= 150:
            self._recalibrate()

    def _recalibrate(self) -> None:
        if self._samples:
            self.floor = float(np.mean(self._samples))
        self._nominal_streak = 0
        self._ticks_since_recal = 0
```

```python
def _opcodes_for(role: str) -> list:
    return _RED_OPCODES if role == "red" else _BLUE_OPCODES


def random_program(role: str, min_len: int = 8, max_len: int = 32) -> list:
    opcodes = _opcodes_for(role)
    target = random.randint(min_len, max_len)
    prog = []
    while len(prog) < target:
        op = random.choice(opcodes)
        prog.append(op)
        if op == "PUSH":
            prog.append(round(random.uniform(0.0, 1.0), 4))
        elif op == "LOOP":
            prog.append(random.randint(1, 4))
    prog = repair_control_flow(prog)
    return prog[:MAX_PROGRAM_LEN]


def _mutation_weights(win_rate: float) -> dict:
    wr = max(0.0, min(1.0, win_rate))
    return {
        "replace":     0.3 + 0.4 * (1.0 - wr),
        "adjust_push": 0.30,
        "insert":      0.20 * (1.0 - wr),
        "delete":      0.10 + 0.10 * wr,
        "swap":        0.10,
    }


def _weighted_choice(weights: dict) -> str:
    keys = list(weights.keys())
    vals = list(weights.values())
    total = sum(vals)
    r = random.uniform(0, total)
    acc = 0.0
    for k, v in zip(keys, vals):
        acc += v
        if r <= acc:
            return k
    return keys[-1]


def crossover(prog_a: list, prog_b: list) -> list:
    mask_a = compute_operand_mask(prog_a)
    valid = [i for i in range(1, len(prog_a)) if not mask_a[i]]
    point = random.choice(valid) if valid else max(1, len(prog_a) // 2)
    child = prog_a[:point] + prog_b[point:]
    child = repair_control_flow(child)
    return child[:MAX_PROGRAM_LEN]


def mutate(program: list, win_rate: float, role: str) -> list:
    opcodes = _opcodes_for(role)
    prog = list(program)
    mask = compute_operand_mask(prog)
    weights = _mutation_weights(win_rate)
    kind = _weighted_choice(weights)

    non_op = [i for i in range(len(prog)) if not mask[i]]
    operand_slots = [i for i in range(len(prog)) if mask[i]]

    if kind == "replace" and non_op:
        i = random.choice(non_op)
        new_op = random.choice(opcodes)
        if prog[i] == "PUSH" and i + 1 < len(prog) and mask[i + 1]:
            prog[i] = new_op
            if new_op == "PUSH":
                prog[i + 1] = round(random.uniform(0.0, 1.0), 4)
            elif new_op == "LOOP":
                prog[i + 1] = random.randint(1, 4)
        else:
            prog[i] = new_op
            if new_op == "PUSH" and i + 1 == len(prog):
                prog.append(round(random.uniform(0.0, 1.0), 4))
            elif new_op == "LOOP" and i + 1 == len(prog):
                prog.append(random.randint(1, 4))

    elif kind == "adjust_push" and operand_slots:
        i = random.choice(operand_slots)
        # Determine if parent is LOOP or PUSH
        parent_i = i - 1
        if parent_i >= 0 and prog[parent_i] == "LOOP":
            prog[i] = max(1, min(8, int(float(prog[i]) + random.gauss(0, 1))))
        else:
            prog[i] = max(0.0, min(10.0, float(prog[i]) + random.gauss(0, 0.1)))

    elif kind == "insert" and len(prog) < MAX_PROGRAM_LEN and non_op:
        i = random.choice(non_op)
        new_op = random.choice(opcodes)
        prog.insert(i, new_op)
        if new_op == "PUSH":
            prog.insert(i + 1, round(random.uniform(0.0, 1.0), 4))
        elif new_op == "LOOP":
            prog.insert(i + 1, random.randint(1, 4))

    elif kind == "delete" and non_op:
        deletable = [i for i in non_op if prog[i] not in _STRUCTURAL]
        if deletable:
            i = random.choice(deletable)
            del prog[i]

    elif kind == "swap" and len(non_op) >= 2:
        i, j = random.sample(non_op, 2)
        prog[i], prog[j] = prog[j], prog[i]

    prog = repair_control_flow(prog)
    return prog[:MAX_PROGRAM_LEN]


def tournament_select(population: list, fitnesses: list, k: int = 3) -> list:
    candidates = random.sample(range(len(population)), k)
    best = max(candidates, key=lambda i: fitnesses[i])
    return population[best]


def init_population(role: str) -> list:
    from shared.genome import BOOTSTRAP_RED, BOOTSTRAP_BLUE
    seed = list(BOOTSTRAP_RED if role == "red" else BOOTSTRAP_BLUE)
    pop = [seed]
    while len(pop) < POPULATION_SIZE:
        pop.append(random_program(role))
    return pop[:POPULATION_SIZE]


def next_generation(population: list, fitnesses: list, win_rate: float, role: str) -> list:
    best_idx = max(range(len(fitnesses)), key=lambda i: fitnesses[i])
    new_pop = [list(population[best_idx])]
    while len(new_pop) < POPULATION_SIZE:
        parent_a = tournament_select(population, fitnesses)
        parent_b = tournament_select(population, fitnesses)
        child = crossover(parent_a, parent_b)
        child = mutate(child, win_rate, role)
        new_pop.append(child)
    return new_pop
```

```python
# host/llm_codegen.py
"""LLM codegen loop — generates new opcode primitives from GA telemetry."""
import ast
import json
import logging
import os
import re
import subprocess
import time
from pathlib import Path
from typing import Optional

_LOG = logging.getLogger(__name__)

ALLOWED_IMPORTS = {
    "os", "subprocess", "json", "random", "time",
    "pathlib", "psutil", "ctypes", "socket", "winreg",
}
FORBIDDEN_CALLS = {"exec", "eval", "compile", "__import__"}

_GENERATED_DIR = Path(__file__).parent.parent / "shared" / "vm" / "generated"
_DEFAULT_BUDGET_USD = float(os.environ.get("CODEGEN_BUDGET_USD", "5.0"))
_spent_usd: float = 0.0


def validate_generated_code(source: str) -> tuple:
    try:
        tree = ast.parse(source)
    except SyntaxError as e:
        return False, f"SyntaxError: {e}"

    for node in ast.walk(tree):
        if isinstance(node, ast.Import):
            for alias in node.names:
                root = alias.name.split(".")[0]
                if root not in ALLOWED_IMPORTS:
                    return False, f"Forbidden import: {alias.name}"
        if isinstance(node, ast.ImportFrom):
            root = (node.module or "").split(".")[0]
            if root not in ALLOWED_IMPORTS:
                return False, f"Forbidden import: {node.module}"
        if isinstance(node, ast.Call):
            func = node.func
            name = None
            if isinstance(func, ast.Name):
                name = func.id
            elif isinstance(func, ast.Attribute):
                name = func.attr
            if name in FORBIDDEN_CALLS:
                return False, f"Forbidden call: {name}"

    return True, "ok"


def parse_generated_file(source: str) -> tuple:
    opcode_m = re.search(r"^# OPCODE: (DO_\w+|SENSE_\w+)", source, re.MULTILINE)
    role_m = re.search(r"^# ROLE: (red|blue)", source, re.MULTILINE)
    fn_m = re.search(r"^def (do_\w+|sense_\w+)\(agent\)", source, re.MULTILINE)

    if not (opcode_m and role_m and fn_m):
        raise ValueError("Generated file missing required header or function signature")

    declared = opcode_m.group(1).lower().replace("do_", "").replace("sense_", "")
    defined = fn_m.group(1).replace("do_", "").replace("sense_", "")
    if declared != defined:
        raise ValueError(f"Opcode name mismatch: {declared!r} vs {defined!r}")

    return opcode_m.group(1), role_m.group(1)


def _next_gen_filename(role: str) -> Path:
    _GENERATED_DIR.mkdir(parents=True, exist_ok=True)
    existing = sorted(_GENERATED_DIR.glob(f"{role}_gen_*.py"))
    n = len(existing) + 1
    return _GENERATED_DIR / f"{role}_gen_{n:03d}.py"


def _read_recent_history(nz: Path, n: int = 20) -> list:
    path = nz / "ga_history.jsonl"
    if not path.exists():
        return []
    lines = path.read_text().strip().splitlines()
    entries = []
    for line in lines[-n:]:
        try:
            entries.append(json.loads(line))
        except Exception:
            pass
    return entries


def _build_prompt(role: str, generation: int, history: list) -> str:
    seqs = []
    for entry in history[-5:]:
        prog = entry.get("red_program" if role == "red" else "blue_program", [])
        if prog:
            seqs.append(str(prog[:10]))
    seq_str = "\n".join(seqs) if seqs else "(none recorded yet)"

    return f"""You are generating a new primitive action for a Windows adversarial AI simulation
running inside Windows Sandbox. The agent is a Python process.

Role: {role}
Current generation: {generation}
Top performing opcode sequences this generation:
{seq_str}

Generate ONE new {role} primitive.

Rules:
- Function signature: def {{snake_case_name}}(agent) -> None
- Allowed imports (must be at module top): os, subprocess, json, random, time, pathlib, psutil, ctypes, socket, winreg
- No exec(), eval(), compile(), __import__(), importlib
- Must complete within 2 seconds
- agent.nz is the neutral zone Path
- agent.t0_pids is the set of baseline PIDs

Output format (exactly):
# OPCODE: DO_SNAKE_CASE_NAME
# ROLE: {role}
# DESCRIPTION: one sentence
def do_snake_case_name(agent) -> None:
    ...
"""


def run_codegen(role: str, generation: int, nz: Path = Path("F:\\neutral_zone")) -> Optional[Path]:
    global _spent_usd

    api_key = os.environ.get("ANTHROPIC_API_KEY")
    if not api_key:
        _LOG.warning("ANTHROPIC_API_KEY not set — skipping codegen")
        return None

    if _spent_usd >= _DEFAULT_BUDGET_USD:
        _LOG.warning("Codegen budget $%.2f exceeded — skipping", _DEFAULT_BUDGET_USD)
        return None

    try:
        import anthropic
    except ImportError:
        _LOG.error("anthropic package not installed — skipping codegen")
        return None

    history = _read_recent_history(nz)
    prompt = _build_prompt(role, generation, history)

    client = anthropic.Anthropic(api_key=api_key)
    try:
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=512,
            messages=[{"role": "user", "content": prompt}],
        )
    except Exception as e:
        _LOG.error("Claude API error: %s", e)
        return None

    # Track spend
    usage = getattr(response, "usage", None)
    if usage:
        input_tokens = getattr(usage, "input_tokens", 0)
        output_tokens = getattr(usage, "output_tokens", 0)
        # Rough estimate: $3/1M input, $15/1M output for Sonnet
        _spent_usd += (input_tokens * 3 + output_tokens * 15) / 1_000_000

    source = response.content[0].text if response.content else ""

    ok, err = validate_generated_code(source)
    if not ok:
        _LOG.error("Generated code failed validation: %s", err)
        return None

    try:
        opcode, file_role = parse_generated_file(source)
    except ValueError as e:
        _LOG.error("Generated file parse error: %s", e)
        return None

    out_path = _next_gen_filename(file_role)
    out_path.write_text(source)
    _LOG.info("Wrote generated op %s to %s", opcode, out_path)

    # Commit the generated file
    try:
        subprocess.run(["git", "add", str(out_path)], check=True)
        subprocess.run(
            ["git", "commit", "-m", f"codegen: add {opcode}"],
            check=True,
        )
    except subprocess.CalledProcessError as e:
        _LOG.warning("git commit failed: %s", e)

    return out_path
```
